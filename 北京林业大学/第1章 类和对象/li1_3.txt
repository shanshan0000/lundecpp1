#include <iostream>
#include <cstring>
using namespace std;

class Student
{
private:
	int number;
	char name[20];
	char sex;
	int age;
public:
	Student();
	Student(int a, char *b, char c, int d);
	void show();
};

Student::Student()
{
	number = 0;
	strcpy(name, "No name");
	sex = '!';
	age = 0;
}

Student::Student(int a, char *b, char c, int d)
{
	number = a;
	strcpy(name, b);
	sex = c;
	age = d;
}

void Student::show()
{
	cout<<number<<'\t'<<name<<'\t'<<sex<<'\t'<<age<<endl;
}

int main()
{
	Student stu1;
	Student stu2(101,"Zhao", 'F', 18);
	stu1.show();
	stu2.show();

	return 0;
}
