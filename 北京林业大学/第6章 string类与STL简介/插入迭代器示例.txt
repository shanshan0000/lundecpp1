#include <iostream>
#include <vector>
#include <iterator>
using namespace std;

int main()
{
    vector<int> vec;
    for(int i=0;i<5;i++)
		vec.push_back(i);

    cout<<"the original array is"<<endl;
    for(int i=0;i<5;i++)
		cout<<vec[i]<<endl;
	
	vector<int>::iterator beg = vec.begin();
    beg+=2;

    insert_iterator<vector<int> > i_itr(vec,beg);
    *i_itr++ = 10;
    *i_itr++ = 20;
    cout<<"array after inserting:"<<endl;
    beg=vec.begin();
    while(beg!=vec.end())
		cout<<*beg++<<endl;

    return 0;
}
